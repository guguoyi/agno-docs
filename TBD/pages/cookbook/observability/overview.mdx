Monitor and debug your agents in production with observability integrations. Agno supports OpenTelemetry-based tracing that works with major observability platforms.

```python
from agno.agent import Agent
from agno.models.openai import OpenAIChat
from openinference.instrumentation.agno import AgnoInstrumentor

# Start tracing
AgnoInstrumentor().instrument()

agent = Agent(model=OpenAIChat(id="gpt-4o"))
agent.print_response("Hello, world!")
# Traces sent to your observability platform
```

## Examples by Platform

### Langfuse

Open-source LLM observability platform.

```python cookbook/91_integrations/observability/langfuse_via_openinference.py
import base64
import os
from agno.agent import Agent
from agno.models.openai import OpenAIChat
from agno.tools.yfinance import YFinanceTools
from openinference.instrumentation.agno import AgnoInstrumentor
from opentelemetry.exporter.otlp.proto.http.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import SimpleSpanProcessor

# Configure Langfuse
LANGFUSE_AUTH = base64.b64encode(
    f"{os.getenv('LANGFUSE_PUBLIC_KEY')}:{os.getenv('LANGFUSE_SECRET_KEY')}".encode()
).decode()
os.environ["OTEL_EXPORTER_OTLP_ENDPOINT"] = "https://cloud.langfuse.com/api/public/otel"
os.environ["OTEL_EXPORTER_OTLP_HEADERS"] = f"Authorization=Basic {LANGFUSE_AUTH}"

# Start tracing
tracer_provider = TracerProvider()
tracer_provider.add_span_processor(SimpleSpanProcessor(OTLPSpanExporter()))
AgnoInstrumentor().instrument(tracer_provider=tracer_provider)

# Use your agent - traces are sent automatically
agent = Agent(
    model=OpenAIChat(id="gpt-4o-mini"),
    tools=[YFinanceTools()],
)

agent.print_response("What is Tesla's stock price?")
```

### Arize Phoenix

AI observability with Arize Phoenix.

```python cookbook/91_integrations/observability/arize_phoenix_via_openinference.py
import phoenix as px
from agno.agent import Agent
from agno.models.openai import OpenAIChat
from openinference.instrumentation.agno import AgnoInstrumentor
from opentelemetry.sdk.trace import TracerProvider

# Start Phoenix
px.launch_app()

tracer_provider = TracerProvider()
AgnoInstrumentor().instrument(tracer_provider=tracer_provider)

agent = Agent(model=OpenAIChat(id="gpt-4o"))
agent.print_response("Explain machine learning")
```

### LangSmith

LangChain's observability platform.

```python cookbook/91_integrations/observability/langsmith_via_openinference.py
import os
from agno.agent import Agent
from agno.models.openai import OpenAIChat
from openinference.instrumentation.agno import AgnoInstrumentor
from opentelemetry.exporter.otlp.proto.http.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import SimpleSpanProcessor

os.environ["OTEL_EXPORTER_OTLP_ENDPOINT"] = "https://api.smith.langchain.com/otel"
os.environ["OTEL_EXPORTER_OTLP_HEADERS"] = f"x-api-key={os.getenv('LANGSMITH_API_KEY')}"

tracer_provider = TracerProvider()
tracer_provider.add_span_processor(SimpleSpanProcessor(OTLPSpanExporter()))
AgnoInstrumentor().instrument(tracer_provider=tracer_provider)
```
